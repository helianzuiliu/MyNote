let finish_time_2=dt.fromObject({weekday:7,hour:22})


if(finish_time_1<now && now<finish_time_2){
	await dv.view("js/scripts/abyss_time_bh3",
	{
		name:name,
		start_time:start_time_2,
		finish_time:finish_time_2
	})
}else{
	await dv.view("js/scripts/abyss_time_bh3",
	{

		finish_time:finish_time_1
	})
}
