const dt = dv.luxon.DateTime
let now = dt.now()
let weekday = now.weekday
let finish_time

if(weekday < 4){
	// 星期三晚上十点结算
	finish_time=dt.fromObject({weekday:3,hour:22})
}else{
    finish_time=dt.fromObject({weekday:7,hour:22})
}

let d = finish_time.weekday - now.weekday
let { h , m } = finish_time.minute - now.minute < 0 ?
	{h:(finish_time.hour-now.hour-1),m:(finish_time.minute-now.minute+60)}
	:{h:(finish_time.hour-now.hour),m:(finish_time.minute-now.minute) }

switch(finish_time.weekday - now.weekday):
case 0:
	dv.paragraph("距离深渊结算还有** "+h+"** 小时 **"+m+"** 分,今天你打深渊了吗")
	break
default:
	dv.paragraph("距离深渊结算还有 **" + d + "** 天,还不用着急")
